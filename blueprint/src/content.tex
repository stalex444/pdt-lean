\chapter{Quantum kinematics from a single posit}

\begin{definition}[State space as a complex place]
  \label{def:qishilbert}
  \lean{PDT.QisHilbert}
  \leanok
  For a real vector space $H$, write $\mathtt{QisHilbert}\,H$ for the type of
  $\mathbb{R}$-linear equivalences $H \simeq \mathbb{C}$. Here $\mathbb{C}$ is the complex
  place of the number field generated by $Q$ (a root of $x^4 - x - 1$).
\end{definition}

\begin{assumption}[The physical identification (the single posit)]
  \label{ass:posit}
  \uses{def:qishilbert}
  A physical one-qubit state space realizes \cref{def:qishilbert}: it is
  $\mathbb{R}$-linearly the complex place of $Q$'s field. This is a \emph{physics} claim,
  not a theorem; it is the one undischarged hypothesis, and everything below is downstream
  of it. (No \texttt{\textbackslash leanok}: a kernel cannot certify a statement about the
  physical world.)
\end{assumption}

\begin{lemma}[Born form is the field norm]
  \label{lem:born_form}
  \lean{PDT.born_form}
  \leanok
  $(z\cdot z^{\dagger}).\mathrm{re} = \mathrm{normSq}\,z = |z|^2$, where $z^{\dagger}$ is the
  Galois conjugation $\mathtt{starRingEnd}\,\mathbb{C}$. The square is forced by
  $[\mathbb{C}:\mathbb{R}] = 2$.
\end{lemma}

\begin{lemma}[Hermitian = Galois-fixed = real (the pointer basis)]
  \label{lem:hermitian}
  \lean{PDT.hermitian_real}
  \leanok
  A Galois-fixed scalar is real: $z^{\dagger} = z \implies \mathrm{im}\,z = 0$.
\end{lemma}

\begin{lemma}[Unitary preserves the Born norm]
  \label{lem:unitary}
  \lean{PDT.unitary_normSq_one}
  \leanok
  $z\cdot z^{\dagger} = 1 \implies \mathrm{normSq}\,z = 1$.
\end{lemma}

\begin{theorem}[The Born rule, derived]
  \label{thm:qm_from_q}
  \lean{PDT.QM_from_Q}
  \leanok
  \uses{ass:posit, lem:born_form}
  Given the posit, for a normalized two-outcome state the Born probabilities equal the
  conjugate-twisted trace form $|z|^2$, are non-negative, and sum to one --- a genuine
  probability rule, with \emph{no} independent Born postulate.
\end{theorem}

\chapter{The golden arithmetic bedrock}

\begin{proposition}[Cubic discriminant and the gauge dimension]
  \label{prop:disc_cubic}
  \lean{PDT.twentythree_weld}
  \leanok
  $-\operatorname{disc}(x^3-x-1) = \dim\mathfrak{su}(3) + \dim\mathfrak{su}(4) = 23$.
\end{proposition}

\begin{proposition}[Quartic discriminant]
  \label{prop:disc_quartic}
  \lean{PDT.disc_quartic}
  \leanok
  $\operatorname{disc}(x^4-x-1) = -283$. Since $283$ is prime
  (\lstinline{PDT.prime_283}) the discriminant is squarefree, so $\mathbb{Z}[Q]$ is the full
  ring of integers and this is also the field discriminant of $\mathbb{Q}(Q)$. It equals the
  determinant of the trace form (\cref{thm:sig_quartic}).
\end{proposition}

\begin{proposition}[Norms]
  \label{prop:norms}
  \lean{PDT.unit_norm_rhoQ}
  \leanok
  $N(\rho) = 1$, $N(Q) = -1$, $N(\rho Q) = 1^4\cdot(-1)^3 = -1$ (the unit norm forcing
  integer exponents), and $N(2Q-1) = -23$ (welding the ``$23$'' to $E_4$'s $2$-torsion).
\end{proposition}

\chapter{The same arithmetic, through Mathlib's number-theory API}

The previous chapter states the norm and discriminant values as elementary integer
identities (closed by \texttt{norm\_num} on the explicit formulas). This chapter
re-derives the \emph{same} facts through Mathlib's genuine number-theory API:
\texttt{Algebra.norm}, \texttt{Algebra.discr}, and \texttt{Algebra.traceForm} over
\texttt{AdjoinRoot}. This is the answer to ``the content is in the names'': the facts
are theorems about the real field-theoretic invariants, not values substituted into a
chosen expression.

\begin{proposition}[Irreducibility via \texttt{minpoly}]
  \label{prop:irred}
  \lean{PDT.cubicQ_irreducible, PDT.quarticQ_irreducible}
  \leanok
  $x^3-x-1$ and $x^4-x-1$ are \texttt{Irreducible} over $\mathbb{Q}$, so the
  \texttt{AdjoinRoot} quotients are genuine number fields and the
  norm / discriminant / trace-form invariants below are the field-theoretic objects
  they claim to be.
\end{proposition}

\begin{proposition}[Field norms via \texttt{Algebra.norm}]
  \label{prop:alg_norm}
  \lean{PDT.norm_Q, PDT.norm_ρ}
  \leanok
  \uses{prop:irred}
  $\operatorname{Algebra.norm}_{\mathbb{Q}}(Q) = -1$ and
  $\operatorname{Algebra.norm}_{\mathbb{Q}}(\rho) = 1$, for $Q,\rho$ the
  \texttt{AdjoinRoot} generators of $\mathbb{Q}[x]/(x^4-x-1)$ and
  $\mathbb{Q}[x]/(x^3-x-1)$ --- the \texttt{Algebra.norm} computation itself, agreeing
  with the elementary values of \cref{prop:norms}.
\end{proposition}

\begin{proposition}[Discriminants via \texttt{Algebra.discr}]
  \label{prop:alg_discr}
  \lean{PDT.discr_cubic, PDT.discr_quartic}
  \leanok
  \uses{prop:irred}
  $\operatorname{Algebra.discr}_{\mathbb{Q}}$ of the power basis equals $-23$ for
  $x^3-x-1$ and $-283$ for $x^4-x-1$ --- the genuine discriminant of the basis,
  agreeing with \cref{prop:disc_cubic,prop:disc_quartic}.
\end{proposition}

\begin{proposition}[The trace form \emph{is} \texttt{Algebra.traceForm}]
  \label{prop:alg_traceform}
  \lean{PDT.traceForm_eq_M4}
  \leanok
  \uses{prop:irred, thm:sig_quartic}
  The Gram matrix $M_4$ behind the $(3,1)$ signature is \emph{exactly}
  $\operatorname{Algebra.traceForm}_{\mathbb{Q}}$ on the power basis of
  $\mathbb{Q}[x]/(x^4-x-1)$: $\operatorname{traceForm}(e_i,e_j) = (M_4)_{ij}$. So the
  signature theorem is a statement about the genuine trace form, not a hand-built matrix.
\end{proposition}

\chapter{Trace-form signatures: 3-space and spacetime}

\begin{theorem}[Spacetime signature $(3,1)$]
  \label{thm:sig_quartic}
  \lean{PDT.signature_3_1}
  \leanok
  The trace form of $\mathbb{Q}[x]/(x^4-x-1)$, in the power basis, is congruent over
  $\mathbb{Q}$ to $\operatorname{diag}(4,4,-\tfrac94,\tfrac{283}{36})$ --- three positive,
  one negative. With $\det = -283 < 0$, by Sylvester's law its signature is $(3,1)$.
\end{theorem}

\begin{theorem}[$3$-space signature $(2,1)$]
  \label{thm:sig_cubic}
  \lean{PDT.signature_2_1}
  \leanok
  The trace form of $\mathbb{Q}[x]/(x^3-x-1)$ is congruent over $\mathbb{Q}$ to
  $\operatorname{diag}(3,2,-\tfrac{23}{6})$ --- two positive, one negative, $\det = -23$.
  By Sylvester's law its signature is $(2,1)$.
\end{theorem}

\chapter{The Bell gap: classical $2$ versus quantum $2\sqrt2$}

\begin{theorem}[Classical CHSH bound]
  \label{thm:chsh_classical}
  \lean{PDT.chsh_comm_le_two}
  \leanok
  In a commutative ordered $*$-ring, any CHSH tuple satisfies
  $A_0B_0 + A_0B_1 + A_1B_0 - A_1B_1 \le 2$ (via Mathlib's
  \lstinline{CHSH_inequality_of_comm}).
\end{theorem}

\begin{theorem}[Tsirelson bound, saturated]
  \label{thm:tsirelson}
  \lean{PDT.chsh_saturates}
  \leanok
  \uses{thm:chsh_classical}
  The Tsirelson bound $2\sqrt2$ (Mathlib's \lstinline{tsirelson_inequality}) is attained:
  an explicit real-Pauli quadruple in $M_2(\mathbb{R})\otimes M_2(\mathbb{R})$ is a genuine
  CHSH tuple whose operator reaches eigenvalue $2\sqrt2$ on the Bell direction. The gap
  $2 < 2\sqrt2$ (\lstinline{PDT.bell_gap}) is strict: commuting observables stop at $2$;
  noncommutativity reaches $2\sqrt2$.
\end{theorem}
